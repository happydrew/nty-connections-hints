import HintsPage from '@components/nytconnectionshints/HintsPage'
import { getLatestGameInfo } from '@lib/data';

export async function getStaticProps() {
    const latestGameInfo = await getLatestGameInfo();
    return {
        props: {
            game_date: latestGameInfo.game_date,
            game_number: latestGameInfo.game_number,
            game_author: latestGameInfo.game_author,
            game_data: JSON.stringify(latestGameInfo.game_data),
            hints: JSON.stringify(latestGameInfo.hints),
            __nextra_dynamic_opts: {
                title: `NYT Connections Hints Today - ${new Date(latestGameInfo.game_date).toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })} | Clues & Answers`,
                frontMatter: {
                    description: `NYT Connections hints today for ${new Date(latestGameInfo.game_date).toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })}, including categorized clues and answers. Play with unlimited mode or check the NYT Connections archive for past puzzles. Unlike Mashable’s Connections hints, this page has a fixed URL, making it easy to access daily without searching.`
                }
            }
        }
    };
}

<HintsPage {...props} has_next={false} />
