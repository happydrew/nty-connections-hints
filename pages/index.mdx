import HintsPage from '@components/nytconnectionshints/HintsPage'
import { getLatestGameInfo } from '@lib/data';

export async function getStaticProps() {
    const latestGameInfo = await getLatestGameInfo();
    return {
        props: {
            game_date: latestGameInfo.game_date,
            game_number: latestGameInfo.game_number,
            game_author: latestGameInfo.game_author,
            game_data: JSON.stringify(latestGameInfo.game_data),
            hints: JSON.stringify(latestGameInfo.hints)
        }
    };
}

<HintsPage {...props} has_next={false} />
